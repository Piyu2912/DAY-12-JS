        function create(){
            //Creating elements
            const ele5 = document.createElement("h1");
            ele5.setAttribute("id","dom");
            ele5.innerText = "It is created using DOM"
            console.log(ele5)
            //Appending the child on screen
            document.getElementById("parent").appendChild(ele5);
        }
